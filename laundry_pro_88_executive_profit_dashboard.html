      Panduan Deploy Online Gratis (Hingga 2050+)

      Anda dapat mengunggah (deploy) file dashboard tunggal index.html ini agar bisa diakses oleh cabang mana saja secara real-time dan aman tanpa biaya sepeser pun.

      Opsi 1: Netlify (Sangat Mudah)
      1. Buat akun di netlify.com.
      2. Masuk ke menu "Add new site" → "Deploy manually".
      3. Seret (drag & drop) file index.html ini ke dalam kotak drop.
      4. Selesai! Dashboard Anda sekarang online dengan URL acak ber-SSL aman (HTTPS).
      Opsi 2: Vercel (Terpopuler)
      1. Buat akun di vercel.com.
      2. Unduh CLI Vercel atau buat repositori di GitHub berisi file index.html.
      3. Hubungkan GitHub Anda ke Vercel dan pilih import repositori tersebut.
      4. Vercel akan mendeploy dashboard Anda dalam waktu kurang dari 10 detik secara gratis.
      Opsi 3: GitHub Pages
      1. Buat repositori baru bernama laundry-dashboard di GitHub Anda.
      2. Unggah file index.html ini ke repositori tersebut.
      3. Buka Settings → Pages, lalu set source ke branch Main/Master.
      4. URL publik Anda akan aktif di username.github.io/laundry-dashboard.

CREATE TABLE `cabang` (
  `id` BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  `nama` VARCHAR(100) NOT NULL UNIQUE,
  `created_at` TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `kategori_keuangan` (
  `id` BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  `nama` VARCHAR(100) NOT NULL,
  `tipe` ENUM('pendapatan', 'pengeluaran') NOT NULL,
  `created_at` TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  UNIQUE KEY `unique_nama_tipe` (`nama`, `tipe`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `log_keuangan_bulanan` (
  `id` BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  `cabang_id` BIGINT UNSIGNED NOT NULL,
  `bulan` TINYINT UNSIGNED NOT NULL,
  `tahun` YEAR NOT NULL,
  `tipe` ENUM('pendapatan', 'pengeluaran') NOT NULL,
  `kategori_id` BIGINT UNSIGNED NOT NULL,
  `nominal` DECIMAL(15,2) NOT NULL,
  `keterangan` TEXT NULL,
  `created_at` TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  CONSTRAINT `fk_log_cabang` FOREIGN KEY (`cabang_id`) REFERENCES `cabang` (`id`) ON DELETE CASCADE,
  CONSTRAINT `fk_log_kategori` FOREIGN KEY (`kategori_id`) REFERENCES `kategori_keuangan` (`id`) ON DELETE RESTRICT,
  INDEX `idx_filter_global` (`cabang_id`, `tahun`, `bulan`),
  INDEX `idx_tipe_transaksi` (`tipe`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

      <?php
namespace App\Http\Controllers\Api;

use App\Http\Controllers\Controller;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\DB;

class ExecutiveFinanceController extends Controller
{
    public function getDashboardData(Request $request)
    {
        $cabangId = $request->query('cabang_id');
        $tahun = $request->query('tahun', date('Y'));

        $query = DB::table('log_keuangan_bulanan')->where('tahun', $tahun);
        if ($cabangId && $cabangId !== 'all') {
            $query->where('cabang_id', $cabangId);
        }

        $ledgerData = $query->get();

        $totalPendapatan = $ledgerData->where('tipe', 'pendapatan')->sum('nominal');
        $totalPengeluaran = $ledgerData->where('tipe', 'pengeluaran')->sum('nominal');
        $netProfit = $totalPendapatan - $totalPengeluaran;
        $profitMargin = $totalPendapatan > 0 ? round(($netProfit / $totalPendapatan) * 100, 2) : 0;

        return response()->json([
            'success' => true,
            'summary' => compact('totalPendapatan', 'totalPengeluaran', 'netProfit', 'profitMargin')
        ]);
    }
}
